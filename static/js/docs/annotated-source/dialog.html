      $(dialogElement).dialog({
        autoOpen: false,
        height: 200,
        width: 350,
        modal: true,
        buttons: {
          'Save Search': handleSaveSearch,
          'Cancel': handleClose
        },
        close: resetError
      });
    };

    var init = function (openEvent, dialogId) {
      dialogElement = dialogId;
      dispatcher.on(openEvent, openDialog);
      createDialog();
